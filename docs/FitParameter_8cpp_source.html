171 bool FitParameter::setFromParsedLine(const ParsedParameterLine& line){
172  const std::vector<std::string>& vsl = line.parsedStrings();
173  if(vsl.size() < 3){
174  if(! _quiet) cout << "ERROR in FitParameter::setFromParsedLine"
175  << " need at least 3 parameters:\n"
176  << " iFix, initial mean"
177  << endl;
178  return false;
179  }
180  if(vsl[0] != _name){
181  if(! _quiet) std::cout << "ERROR in FitParameter::setFromParsedLine"
182  << " wrong parameter name. My name is " << _name
183  << " the line's first element is " << vsl[0]
184  << endl;
185  return false;
186  }
187  unsigned int start=0;
188  // if(vsl.size() >=7) ++start; // 1st parameter is index number
189  // above was for compatibility with Mikhail's code, not needed anymore
190 
191  if(vsl.size() > ++start) stringToAnything(vsl[start], _iFixInit);
192  if(vsl.size() > ++start) stringToAnything(vsl[start], _meanInit);
193  if(vsl.size() > ++start) stringToAnything(vsl[start], _stepInit);
194  if(vsl.size() > ++start) stringToAnything(vsl[start], _minInit);
195  if(vsl.size() > ++start) stringToAnything(vsl[start], _maxInit);
196  initToResult();
197 
198  return true;
199 }
